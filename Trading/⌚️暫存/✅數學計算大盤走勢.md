import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import chi2_contingency, f_oneway
import scipy.stats as st

# Setup Data
df = pd.read_csv('大盤交易資料.csv')
cols = ['交易日期', '開盤', '最高', '最低', '收盤']
df = df[cols].copy()
df.columns = ['Date', 'Open', 'High', 'Low', 'Close']
for c in ['Open', 'High', 'Low', 'Close']:
    df[c] = df[c].astype(str).str.replace(',', '').astype(float)
df['Date'] = pd.to_datetime(df['Date'])
df = df.sort_values('Date').reset_index(drop=True)

# ----- Part 1: Markov Chain -----
df['State'] = (df['Close'] > df['Open']).astype(int) # 1: 漲, 0: 跌
df['State_1'] = df['State'].shift(1)
df['State_2'] = df['State'].shift(2)

# Conditional prob: P(Day3=1 | Day1=1, Day2=1)
# Note: Since sorted chronologically, Day1 is t-2, Day2 is t-1, Day3 is t.
# Wait, "連續 2 天出現『漲』的情況下，第 3 天出現『漲』": 
# This means if t-2 == 1 and t-1 == 1, what is prob of t == 1?
cond_mask = (df['State_2'] == 1) & (df['State_1'] == 1)
total_cond = cond_mask.sum()
prob_3rd_up = (df.loc[cond_mask, 'State'] == 1).sum() / total_cond if total_cond > 0 else 0

# 3-day sequences
seq_df = df.dropna(subset=['State_2', 'State_1', 'State']).copy()
seq_df['Seq'] = seq_df['State_2'].astype(int).astype(str) + seq_df['State_1'].astype(int).astype(str) + seq_df['State'].astype(int).astype(str)
seq_counts = seq_df['Seq'].value_counts()
seq_probs = seq_counts / len(seq_df)

print("=== Part 1: Markov Chain ===")
print(f"P(Day3=Up | Day1=Up, Day2=Up) = {prob_3rd_up:.4f}")
print("Sequence Probs:\n", pd.DataFrame({'Freq': seq_counts, 'Prob': seq_probs}))


# ----- Part 2: Monte Carlo Simulation -----
df['LogRet'] = np.log(df['Close'] / df['Close'].shift(1))
mu = df['LogRet'].mean()
sigma = df['LogRet'].std()

S0 = df['Close'].iloc[-1]
days = 10
simulations = 10000

np.random.seed(42)
Z = np.random.normal(0, 1, (days, simulations))
paths = np.zeros((days + 1, simulations))
paths[0] = S0
for t in range(1, days + 1):
    paths[t] = paths[t-1] * np.exp((mu - 0.5 * sigma**2) + sigma * Z[t-1])

final_prices = paths[-1]
ci_lower = np.percentile(final_prices, 2.5)
ci_upper = np.percentile(final_prices, 97.5)

plt.figure(figsize=(10, 6))
counts, bins, patches = plt.hist(final_prices, bins=50, color='skyblue', edgecolor='black', alpha=0.7)
max_bin_idx = np.argmax(counts)
best_interval = (bins[max_bin_idx], bins[max_bin_idx+1])
plt.axvline(ci_lower, color='red', linestyle='dashed', linewidth=2, label=f'2.5%: {ci_lower:.2f}')
plt.axvline(ci_upper, color='red', linestyle='dashed', linewidth=2, label=f'97.5%: {ci_upper:.2f}')
plt.title('Monte Carlo Simulation: 10-Day Final Price Distribution')
plt.xlabel('Price')
plt.ylabel('Frequency')
plt.legend()
plt.savefig('monte_carlo.png')
plt.close()

print("\n=== Part 2: Monte Carlo ===")
print(f"Mu: {mu:.6f}, Sigma: {sigma:.6f}")
print(f"Current Price S0: {S0}")
print(f"Highest Probability Interval: {best_interval[0]:.2f} - {best_interval[1]:.2f}")
print(f"95% Confidence Interval: [{ci_lower:.2f}, {ci_upper:.2f}]")


# ----- Part 3: Feature Engineering & Stat Test -----
def compute_rsi(data, window=14):
    diff = data.diff(1).dropna()
    gain = (diff.where(diff > 0, 0)).rolling(window=window).mean()
    loss = (-diff.where(diff < 0, 0)).rolling(window=window).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    return rsi

df['RSI'] = compute_rsi(df['Close'], 14)
df['RSI_Bin'] = pd.cut(df['RSI'], bins=range(0, 110, 10), labels=[f"{i}-{i+10}" for i in range(0, 100, 10)])
df['Next_State'] = df['State'].shift(-1)
df['Next_Return'] = df['Close'].shift(-1) / df['Close'] - 1

rsi_df = df.dropna(subset=['RSI_Bin', 'Next_State', 'Next_Return']).copy()
rsi_stats = rsi_df.groupby('RSI_Bin').agg(
    Count=('Next_State', 'count'),
    WinRate=('Next_State', 'mean'),
    ExpectedReturn=('Next_Return', 'mean')
).dropna()

# Chi-Square test for Win Rate
contingency = pd.crosstab(rsi_df['RSI_Bin'], rsi_df['Next_State'])
chi2, p_val_chi2, dof, _ = chi2_contingency(contingency)

# ANOVA for Expected Return
groups = [group['Next_Return'].values for name, group in rsi_df.groupby('RSI_Bin') if len(group) > 0]
f_stat, p_val_anova = f_oneway(*groups)

print("\n=== Part 3: RSI Stats ===")
print(rsi_stats)
print(f"Chi-square p-value for Win Rate: {p_val_chi2:.4f}")
print(f"ANOVA p-value for Expected Return: {p_val_anova:.4f}")


# ----- Part 4: K-Line Patterns -----
df['Body'] = np.abs(df['Close'] - df['Open'])
df['Upper'] = df['High'] - np.maximum(df['Open'], df['Close'])
df['Lower'] = np.minimum(df['Open'], df['Close']) - df['Low']

# 1. Hammer (Long Lower Shadow)
# 2. Shooting Star (Long Upper Shadow)
# 3. Doji (Cross)
df['Pattern'] = 'None'
df.loc[(df['Lower'] > 2 * df['Body']) & (df['Upper'] < df['Body']), 'Pattern'] = 'Hammer'
df.loc[(df['Upper'] > 2 * df['Body']) & (df['Lower'] < df['Body']), 'Pattern'] = 'ShootingStar'
df.loc[(df['Body'] <= (df['High'] - df['Low']) * 0.1) & (df['High'] > df['Low']), 'Pattern'] = 'Doji'

# Calculate future returns
df['Ret_1d'] = df['Close'].shift(-1) / df['Close'] - 1
df['Ret_3d'] = df['Close'].shift(-3) / df['Close'] - 1
df['Ret_5d'] = df['Close'].shift(-5) / df['Close'] - 1

patterns = ['Hammer', 'ShootingStar', 'Doji']
pat_stats = []
for p in patterns:
    sub = df[df['Pattern'] == p].copy()
    if len(sub) == 0: continue
    
    r1, r3, r5 = sub['Ret_1d'].dropna(), sub['Ret_3d'].dropna(), sub['Ret_5d'].dropna()
    
    # Win rate
    wr1 = (r1 > 0).mean()
    
    # P/L Ratio for 1d
    avg_win = r1[r1 > 0].mean() if len(r1[r1 > 0]) > 0 else 0
    avg_loss = np.abs(r1[r1 < 0].mean()) if len(r1[r1 < 0]) > 0 else 1
    pl_ratio = avg_win / avg_loss if avg_loss != 0 else 0
    
    pat_stats.append({
        'Pattern': p, 'Count': len(sub),
        'AvgRet_1d': r1.mean(), 'AvgRet_3d': r3.mean(), 'AvgRet_5d': r5.mean(),
        'WinRate_1d': wr1, 'PL_Ratio_1d': pl_ratio
    })

print("\n=== Part 4: K-Line Patterns ===")
print(pd.DataFrame(pat_stats))

plt.figure(figsize=(8, 6))
plot_df = df[df['Pattern'].isin(patterns)].dropna(subset=['Ret_1d'])
sns.boxplot(x='Pattern', y='Ret_1d', data=plot_df)
plt.title('1-Day Return Distribution by K-Line Pattern')
plt.axhline(0, color='red', linestyle='--')
plt.savefig('kline_patterns.png')
plt.close()
