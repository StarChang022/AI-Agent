import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats
import warnings
warnings.filterwarnings('ignore')

# 1. Data Cleaning
df = pd.read_csv("大盤交易資料.csv")
df['交易日期'] = pd.to_datetime(df['交易日期'])
df = df.sort_values('交易日期').reset_index(drop=True)

# Filter last 5 years
max_date = df['交易日期'].max()
five_years_ago = max_date - pd.DateOffset(years=5)
df = df[df['交易日期'] >= five_years_ago].reset_index(drop=True)

cols_to_clean = ['開盤', '最高', '最低', '收盤', '漲跌']
for col in cols_to_clean:
    df[col] = df[col].astype(str).str.replace(',', '', regex=False).astype(float)

results = {}

# -----------------
# Task 1: Markov Chain
# -----------------
# 1. State: Up (Close > Open), Down (Close <= Open)
df['State'] = np.where(df['收盤'] > df['開盤'], 'U', 'D')

# 2 & 3. 3-day sequence
df['State_1'] = df['State'].shift(2)
df['State_2'] = df['State'].shift(1)
df['State_3'] = df['State']

seq_df = df.dropna(subset=['State_1', 'State_2', 'State_3']).copy()
seq_df['Seq'] = seq_df['State_1'] + seq_df['State_2'] + seq_df['State_3']

# Freq & Prob
seq_counts = seq_df['Seq'].value_counts()
seq_probs = seq_df['Seq'].value_counts(normalize=True)
markov_table = pd.DataFrame({'Frequency': seq_counts, 'Probability': seq_probs}).sort_index()
markov_table.to_csv("markov_table.csv")
print("Markov Table saved.")

# P(Day3=U | Day1=U, Day2=U)
uu_count = len(seq_df[(seq_df['State_1'] == 'U') & (seq_df['State_2'] == 'U')])
uuu_count = len(seq_df[(seq_df['State_1'] == 'U') & (seq_df['State_2'] == 'U') & (seq_df['State_3'] == 'U')])
prob_u_given_uu = uuu_count / uu_count if uu_count > 0 else 0
print(f"P(U | UU) = {prob_u_given_uu}")

# -----------------
# Task 2: Monte Carlo
# -----------------
df['Log_Return'] = np.log(df['收盤'] / df['收盤'].shift(1))
mu = df['Log_Return'].mean()
sigma = df['Log_Return'].std()

S0 = df['收盤'].iloc[-1]
days = 10
simulations = 10000

np.random.seed(42)
# GBM formula: S_t = S_0 * exp((mu - 0.5*sigma^2)*t + sigma * W_t)
# Daily path
dt = 1
paths = np.zeros((days + 1, simulations))
paths[0] = S0

for t in range(1, days + 1):
    Z = np.random.standard_normal(simulations)
    paths[t] = paths[t-1] * np.exp((mu - 0.5 * sigma**2) * dt + sigma * np.sqrt(dt) * Z)

final_prices = paths[-1]
ci_lower = np.percentile(final_prices, 2.5)
ci_upper = np.percentile(final_prices, 97.5)

# Plot MC
plt.figure(figsize=(10, 6))
plt.hist(final_prices, bins=50, color='skyblue', edgecolor='black', alpha=0.7)
plt.axvline(ci_lower, color='red', linestyle='dashed', linewidth=2, label=f'95% CI Lower ({ci_lower:.2f})')
plt.axvline(ci_upper, color='red', linestyle='dashed', linewidth=2, label=f'95% CI Upper ({ci_upper:.2f})')
plt.title('Monte Carlo Simulation: Price Distribution After 10 Days')
plt.xlabel('Price')
plt.ylabel('Frequency')
plt.legend()
plt.savefig('monte_carlo_dist.png')
plt.close()

# Most probable bin
hist, bins = np.histogram(final_prices, bins=50)
max_bin_idx = np.argmax(hist)
mode_range = (bins[max_bin_idx], bins[max_bin_idx+1])
print(f"MC Most probable range: {mode_range}")
print(f"MC 95% CI: ({ci_lower}, {ci_upper})")

# -----------------
# Task 3: Feature Eng & Stats
# -----------------
def calc_rsi(data, periods=14):
    delta = data.diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=periods).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=periods).mean()
    rs = gain / loss
    return 100 - (100 / (1 + rs))

df['RSI_14'] = calc_rsi(df['收盤'], 14)
df['RSI_Bin'] = pd.cut(df['RSI_14'], bins=range(0, 101, 10), labels=[f"{i}-{i+10}" for i in range(0, 100, 10)])
df['Next_Day_Red'] = (df['收盤'].shift(-1) > df['開盤'].shift(-1)).astype(int)
df['Next_Day_Return'] = (df['收盤'].shift(-1) - df['收盤']) / df['收盤']

rsi_df = df.dropna(subset=['RSI_Bin', 'Next_Day_Red', 'Next_Day_Return'])

rsi_stats = rsi_df.groupby('RSI_Bin').agg(
    Count=('Next_Day_Red', 'count'),
    Win_Rate=('Next_Day_Red', 'mean'),
    Expected_Return=('Next_Day_Return', 'mean')
).dropna()
rsi_stats.to_csv("rsi_stats.csv")

# Chi-Square Test for Win Rate across bins
contingency = pd.crosstab(rsi_df['RSI_Bin'], rsi_df['Next_Day_Red'])
chi2, p_val_chi2, _, _ = stats.chi2_contingency(contingency)
print(f"RSI P-value (Chi2): {p_val_chi2}")

# -----------------
# Task 4: K-Line Pattern
# -----------------
df['Body'] = np.abs(df['開盤'] - df['收盤'])
df['Upper_Shadow'] = df['最高'] - np.maximum(df['開盤'], df['收盤'])
df['Lower_Shadow'] = np.minimum(df['開盤'], df['收盤']) - df['最低']
df['HL_Range'] = df['最高'] - df['最低']

# Pattern rules
cond_doji = df['Body'] <= (df['HL_Range'] * 0.1)
cond_hammer = (df['Lower_Shadow'] > 2 * df['Body']) & (df['Upper_Shadow'] <= df['Body']) & (~cond_doji)
cond_shooting_star = (df['Upper_Shadow'] > 2 * df['Body']) & (df['Lower_Shadow'] <= df['Body']) & (~cond_doji)

df['Pattern'] = 'None'
df.loc[cond_doji, 'Pattern'] = 'Doji'
df.loc[cond_hammer, 'Pattern'] = 'Hammer'
df.loc[cond_shooting_star, 'Pattern'] = 'ShootingStar'

for d in [1, 3, 5]:
    df[f'Ret_{d}d'] = (df['收盤'].shift(-d) - df['收盤']) / df['收盤']

patterns = ['Doji', 'Hammer', 'ShootingStar']
pattern_stats = []
plot_data = {}

for p in patterns:
    p_df = df[df['Pattern'] == p].dropna(subset=['Ret_1d', 'Ret_3d', 'Ret_5d'])
    if len(p_df) > 0:
        ret1 = p_df['Ret_1d']
        ret3 = p_df['Ret_3d']
        ret5 = p_df['Ret_5d']
        
        plot_data[p] = ret5 # Store 5d returns for boxplot
        
        # P/L Ratio based on 5d
        wins = ret5[ret5 > 0]
        losses = ret5[ret5 <= 0]
        avg_win = wins.mean() if len(wins) > 0 else 0
        avg_loss = abs(losses.mean()) if len(losses) > 0 else 0
        pl_ratio = avg_win / avg_loss if avg_loss != 0 else np.inf
        
        pattern_stats.append({
            'Pattern': p,
            'Count': len(p_df),
            'Avg_Ret_1d': ret1.mean(),
            'Avg_Ret_3d': ret3.mean(),
            'Avg_Ret_5d': ret5.mean(),
            'Win_Rate_5d': len(wins) / len(p_df),
            'PL_Ratio': pl_ratio
        })

pd.DataFrame(pattern_stats).to_csv("pattern_stats.csv", index=False)

# Boxplot
if plot_data:
    plt.figure(figsize=(10, 6))
    plt.boxplot([plot_data[p] for p in plot_data.keys()], labels=plot_data.keys())
    plt.axhline(0, color='red', linestyle='dashed', linewidth=1)
    plt.title('Return Distribution (5 Days Later) by K-Line Pattern')
    plt.ylabel('Return')
    plt.savefig('pattern_boxplot.png')
    plt.close()

print("All tasks completed.")
